ztree属性组组件的应用————两个ztree的联动

在这做后台的经历可以说大开眼界，每个项目都能看到不同的插件，光怪陆离不一而足⊙﹏⊙b。现在要说的是ztree，一个树形表单插件。api地址：http://www.ztree.me/v3/api.php
需求是，两个数据相同的树，左边的勾选一条，右边相应的条目要隐藏；反之，取消勾选时，显示出已隐藏的对应条目，如果有父级，则连同父级也显示出来

<!DOCTYPE html>
<HTML>
<HEAD>
<TITLE>zTree树分层显示，分层联动</TITLE>
<meta http-equiv="content-type" content="text/html; charset=UTF-8">
<link rel="stylesheet" href="../../css/demo.css" type="text/css">
<link rel="stylesheet" href="../../css/zTreeStyle/zTreeStyle.css" type="text/css">
<script type="text/javascript" src="../../js/jquery-1.4.4.min.js"></script>
<script type="text/javascript" src="../../js/jquery.ztree.core.js"></script>
<script type="text/javascript" src="../../js/jquery.ztree.excheck.js"></script>
<script type="text/javascript" src="../../js/jquery.ztree.exhide.js"></script>

<!-- <script type="text/javascript" src="../../js/jquery.ztree.all-3.4.js"></script> -->
</HEAD>
<BODY>
<ul id="tree1" name="tree1" class="ztree" style="float:left"></ul>
<ul id="tree2"  name="tree2" class="ztree" style="float:left"></ul>
<!-- <ul id="tree3"  name="tree3" class="ztree" style="float:left"></ul> -->
<script type="text/javascript">
        var nodes =  [
                       {name:"test1", open:true, children:[
                          {name:"test1_1"}, {name:"test1_2"}]},
                       {name:"test2", open:true, children:[
                          {name:"test2_1"}, {name:"test2_2"}]}
                       ];



        var settings = {
            check: {
                enable: true,
                chkboxType: { "Y" : "ps", "N" : "ps" }
            },
            data: {
                simpleData: {
                    enable: true
                }
            },
            check:{
                enable: true,
               // nocheckInherit: false,
                chkboxType: { "Y" : "ps", "N" : "ps" }
            },
            callback: {
                onCheck: zTreeOnCheck
            }
        };

        function zTreeOnCheck(e, treeId, treeNode){
            var tree1 = $.fn.zTree.getZTreeObj("tree1");//tree2的ul的id
            var tree2 = $.fn.zTree.getZTreeObj("tree2");//tree2的ul的id

            if(treeNode.checked){
                tree1.hideNode(tree1.getNodeByParam("name", treeNode.name));
                // tree1.selectNode(tree1.getNodeByParam("name", treeNode.name), true);
            }
            else{
                var node = tree1.getNodeByParam("name", treeNode.name);
                tree1.setChkDisabled(node, false);
                tree1.showNode(node);
                var thisNode = node;

                while(node.getParentNode() && node.getParentNode().isHidden == true){
                    node = node.getParentNode();
                }
                if(thisNode == node){
                    return;
                }
                tree1.showNode(node);
                checkChildren(node, thisNode);
                tree1.hideNodes(tree1.getSelectedNodes());
            }
        }

        function checkChildren(node, thisNode){
            console.log(node);
            if(node.children){
                for(var i in node.children){
                    checkChildren(node.children[i], thisNode);    
                } 
            }
            else{
                var tree1 = $.fn.zTree.getZTreeObj("tree1");//tree2的ul的id
                if(node.name !== thisNode.name){
                    tree1.hideNode(node);    
                }

            }
        }

        $(document).ready(function () {
            $.fn.zTree.init($("#tree1"), settings, nodes);
            $.fn.zTree.init($("#tree2"), settings, nodes);

            // 初始化tree1之前为其绑定点击时的函数为tree1OnClick
            // $('#tree2').click(function(e){
            //     tree1OnClick(e, $(this).attr('id'), $(this));
            // })

        });
</script>

</BODY>
</HTML>

要注意的是，在用checkNode或者selectNode这里方法时，参数直接传形如{‘name’: ‘test’}的数据不起作用，要用getNodeByParam()获取结点json
